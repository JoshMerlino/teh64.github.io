function tabs(self){
	try{
		var elementName = self.getAttribute("name");
		var tabElement = document.getElementsByName(elementName);
		var i = 0;
		while(i < tabElement.length){
			tabElement[i].setAttribute("active","false");
			i++
		}
		self.setAttribute("active","true");
	} catch (error) {
		alert(error)
	}
}
